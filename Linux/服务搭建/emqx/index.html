    docker run -itd \
    # 容器节点名称(修改)
    --name emqx01 \
    # EMQX名称和容器节点名称保持一致(修改)
    -e EMQX_NAME=emqx01 \
    # 本机IP 地址(修改)
    -e EMQX_HOST=10.130.21.233 \
    # 前端面板登录用户名(修改)
    -e EMQX_DASHBOARD__DEFAULT_USER__LOGIN=admin \
    # 前端面板登录用户密码(修改)
        -e EMQX_DASHBOARD__DEFAULT_USER__PASSWORD=admin123456 \
    -e EMQX_CLUSTER__DISCOVERY=static \
    -e EMQX_LOADED_PLUGINS="emqx_management,emqx_auth_mnesia,emqx_recon,emqx_retainer,emqx_dashboard" \
    -e EMQX_ALLOW_ANONYMOUS=false \
    # 集群节点(修改)
        -e EMQX_CLUSTER__STATIC__SEEDS=[emqx01@10.130.21.233,emqx02@10.130.21.234,emqx03@10.130.21.235] \
         # 授权访问mqtt 1883端口用户名(修改)
    -e EMQX_AUTH__USER__1__USERNAME=admin \
        # 授权访问mqtt 1883端口用户密码(修改)
    -e EMQX_AUTH__USER__1__PASSWORD=admin123 \
        -v /etc/localtime:/etc/localtime:ro \
        --network=host \
    emqx/emqx

    user  root;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;
worker_rlimit_nofile 655350;

events {
    worker_connections  65535;
}


stream{
        log_format proxy '$time_iso8601 '
        '$remote_addr -> :$server_port -> $upstream_addr'
        ' "$protocol" "$status"'
        ' "$upstream_connect_time"';

        upstream emqx_servers {
                server 10.130.21.233:1883  max_fails=3 fail_timeout=30s;
                server 10.130.21.234:1883  max_fails=3 fail_timeout=30s;
                server 10.130.21.235:1883  max_fails=3 fail_timeout=30s;
        }

        server {
                 listen 1883;
                 proxy_pass emqx_servers;
                 access_log /var/log/nginx/access.log proxy;
                 error_log /var/log/nginx/error.log;
        }
}
